###Structure Name###
$tagDTPRANGE

###Description###
指定一个协调世界时(UTC)日期与时间

###Syntax###
Global Const $tagDTPRANGE = "word MinYear;" & _
                            "word MinMonth;" & _
                            "word MinDOW;" & _
                            "word MinDay;" & _
                            "word MinHour;" & _
                            "word MinMinute;" & _
                            "word MinSecond;" & _
                            "word MinMSecond;" & _
                            "word MaxYear;" & _
                            "word MaxMonth;" & _
                            "word MaxDOW;" & _
                            "word MaxDay;" & _
                            "word MaxHour;" & _
                            "word MaxMinute;" & _
                            "word MaxSecond;" & _
                            "word MaxMSecond;" & _
                            "bool MinValid;" & _
                            "bool MaxValid"

###Fields###
@@ParamTable@@
MinYear
	年的最小值
MinMonth
	月的最小值
MinDOW
	周内日的最小值
MinDay
	日的最小值
MinHour
	时的最小值
MinMinute
	分的最小值
MinSecond
	秒的最小值
MinMSecond
	毫秒的最小值
MaxYear
	年的最大值
MaxMonth
	月的最大值
MaxDOW
	周内日的最大值
MaxDay
	日的最大值
MaxHour
	时的最大值
MaxMinute
	分的最大值
MaxSecond
	秒的最大值
MaxMSecond
	毫秒的最大值
MinValid
	如设置为 返回 True, 最小值有效
MaxValid
	如设置为 返回 True, 最大值有效
@@End@@
