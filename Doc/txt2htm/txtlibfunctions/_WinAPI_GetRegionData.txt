#Include <WinAPIEx.au3>
_WinAPI_GetRegionData ( $hRgn, ByRef $tRGNDATA )

$hRgn

$tRGNDATA
	接收区域逻辑单位数据的$tagRGNDATA结构
@@End@@

###ReturnValue###
