###Structure Name###
$tagMCSELRANGE

###Description###
指定协调世界时(UTC)中的日期和时间

###Syntax###
Global Const $tagMCSELRANGE = "short MinYear;short MinMonth;short MinDOW;short MinDay;short MinHour;short MinMinute;short MinSecond;" & _
		"short MinMSeconds;short MaxYear;short MaxMonth;short MaxDOW;short MaxDay;short MaxHour;short MaxMinute;short MaxSecond;" & _
		"short MaxMSeconds"

###Fields###
@@ParamTable@@
MinYear
	年(最小值)
MinMonth
	月(最小值)
MinDOW
	一周中第几天(最小值)
MinDay
	日(最小值)
MinHour
	时(最小值)
MinMinute
	分(最小值)
MinSecond
	秒(最小值)
MinMSeconds
	毫秒(最小值)
MaxYear
	年(最大值)
MaxMonth
	月(最大值)
MaxDOW
	一周中第几天(最大值)
MaxDay
	日(最大值)
MaxHour
	时(最大值)
MaxMinute
	分(最大值)
MaxSecond
	秒(最大值)
MaxMSeconds
	毫秒(最大值)
@@End@@
