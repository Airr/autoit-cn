###Structure Name###
$tagMCMONTHRANGE

###Description###
获取日期控件中显示日期的高低界限信息

###Syntax###
Global Const $tagMCMONTHRANGE = "short MinYear;" & _
                                "short MinMonth;" & _
                                "short MinDOW;" & _
                                "short MinDay;" & _
                                "short MinHour;" & _
                                "short MinMinute;" & _
                                "short MinSecond;" & _
                                "short MinMSeconds;" & _
                                "short MaxYear;" & _
                                "short MaxMonth;" & _
                                "short MaxDOW;" & _
                                "short MaxDay;" & _
                                "short MaxHour;" & _
                                "short MaxMinute;" & _
                                "short MaxSecond;" & _
                                "short MaxMSeconds;" & _
                                "short Span"

###Fields###
@@ParamTable@@
MinYear
	年 （最小值 ） 
MinMonth
	月 （最小值 ） 
MinDOW
	周中日 （最小值 ） 
MinDay
	日 （最小值 ） 
MinHour
	时 （最小值 ） 
MinMinute
	分 （最小值 ） 
MinSecond
	秒 （最小值 ） 
MinMSeconds
	毫秒 （最大值 ） 
MaxYear
	年 （最大值 ） 
MaxMonth
	月 （最大值 ） 
MaxDOW
	周中日 （最大值 ） 
MaxDay
	日 （最大值 ） 
MaxHour
	时 （最大值 ） 
MaxMinute
	分 （最大值 ） 
MaxSecond
	秒 （最大值 ） 
MaxMSeconds
	毫秒 （最大值 ） 
Span
	日期控件中高低界限的跨度范围
@@End@@
