_GUICtrlToolbar_SetButtonText($hWnd, $iCommandID, $sText)

###Parameters###
@@ParamTable@@
$hWnd
	控件句柄
$iCommandID
	按钮命令 ID
$sText
	按钮文本
@@End@@
